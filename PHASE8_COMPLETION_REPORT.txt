================================================================================
COMMAND LINE OPTIONS
================================================================================

REQUIRED:
  --log PATH              ArduPilot .bin log file
  --frame TYPE            Vehicle frame (quad_x, quad_plus, hexa_x, octo_x)
  --mass FLOAT            Vehicle mass in kg
  --output PATH           Output .parm file

OPTIONAL PRIORS:
  --mass-std FLOAT        Mass uncertainty (default: 0.05 kg)
  --arm-length FLOAT      Motor arm length (default: 0.165 m)
  --arm-length-std FLOAT  Arm uncertainty (default: 0.005 m)
  --prop-diameter FLOAT   Prop diameter (default: 5.1 in)
  --motor-kv FLOAT        Motor KV (default: 2300)

OPTIONAL OUTPUT:
  --report PATH           Generate JSON report
  --verbose               Detailed progress
  --excitation-check-only Check excitation only

BUILT-IN:
  --version               Show version
  --help                  Show help

================================================================================
USAGE EXAMPLES
================================================================================

BASIC:
  python -m cli.sysid \
      --log flight.bin \
      --frame quad_x \
      --mass 1.2 \
      --output sitl_params.parm

FULL OPTIONS:
  python -m cli.sysid \
      --log flight.bin \
      --frame quad_x \
      --mass 1.2 \
      --mass-std 0.05 \
      --arm-length 0.165 \
      --prop-diameter 5.1 \
      --motor-kv 2300 \
      --output sitl_params.parm \
      --report results.json \
      --verbose
